# Ti.Systembars


## Usage

```javascript
var SysBars = require("ti.systembars");
win.addEventListener("open",function(){
	SysBars.hideSystembars();
	SysBars.setStatusBarColor("#ff0000");
	SysBars.setNavigationBarColor("#88ff0000");
	
});
var SysBars = require("ti.systembars");
SysBars.hideSystemNavigationBar();
```
